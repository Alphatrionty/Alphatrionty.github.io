var islandPerimeter = function(grid) {
    alone = 0;
    together = 0;
    for(let i = 0;i < grid.length;i++) {
        for(let j = 0;j < grid[0].length;j++) {
            if(grid[i][j] === 1) {
                alone++;
                if(i < grid.length - 1 && grid[i+1][j] === 1) {
                    together++;
                }
                if(j < grid[0].length - 1 && grid[i][j+1] === 1) {
                    together++;
                }
            }
        }
    }
    return alone * 4 - together * 2;
};
