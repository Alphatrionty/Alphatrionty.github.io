var addStrings = function(num1, num2) {
  let i = num1.length - 1
  let j = num2.length - 1
  let add = 0
  let res = []
  while(i >= 0 || j >= 0 || add != 0) {
    let x = i >= 0 ? num1.charAt(i)-'0' : 0
    let y = j >= 0 ? num2.charAt(j)-'0' : 0
    let result = x + y +add
    res.push(result % 10)

    j--
  }
  return res.reverse().join('')
};

var romanToInt = function(s) {
    const map = {
        I: 1,
        V: 5,
        X: 10,
        L: 50,
        C: 100,
        D: 500,
        M: 1000,
        IV: 4,
        IX: 9,
        XL: 40,
        XC: 90,
        CD: 400,
        CM: 900
    }
    let result = 0;
    for(let i = 0;i < s.length;) {
        if(i + 1< s.length && map[s.substring(i, i+2)]) {
            result += map[s.substring(i, i+2)];
            i +=2;
        }else {
            result += map[s.substring(i, i+1)];
            i++;
        }
    }
    return result;
};

var longestCommonPrefix = function(strs) {
    let res = strs[0];
    if(strs.length === 1) {
        return res;
    }

            res = res.slice(0, res.length - 1);
        }
    }
    return res;

var lengthOfLastWord = function(s) {
    let end = s.length - 1;
    while(end > 0 && s[end] === ' ') end--;
    if(end < 0) return 0;

    return end - start;
