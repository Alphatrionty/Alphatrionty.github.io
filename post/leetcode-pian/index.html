var reverseList = function(head) {
     let prev = null
     let curr = head
     let next = head
     while(curr !== null) {
        next = curr.next
        curr.next = prev
        prev = curr
        curr = next
        }
        return prev
};

🔒415.字符串相加

思路为小学的竖式加法，将数组的每一项进行加法运算，进位保存在add

先设置i，j保存两个数组的长度减一作为下标值
一个进位值add，初始为0
一个结果数组res，用来保存每次竖式加法后的个位值
用一个while循环，条件为数组num1还没循环完 || 数组num2还没循环完 || add不为0
从数组最后一位开始执行竖式加法，为空则补0
设置一个result(x+y+add)为每次竖式加法后的结果
个位作为结果存进结果数组res
十位作为进位赋值给add
i--，j--
进行下一次竖式加法
最后将res反转，再转成字符串

var addStrings = function(num1, num2) {
  let i = num1.length - 1
  let j = num2.length - 1
  let add = 0
  let res = []
  while(i >= 0 || j >= 0 || add != 0) {
    let x = i >= 0 ? num1.charAt(i)-'0' : 0
    let y = j >= 0 ? num2.charAt(j)-'0' : 0
    let result = x + y +add
    res.push(result % 10)
    add = Math.floor(result / 10)
    i--
    j--
  }
  return res.reverse().join('')
};

🔒88.合并两个有序数组

nums1长度为两数组元素之和
此题思路为从数组最后一个开始比较，谁大将谁放进nums1的后面，依次往前放
如果数组元素不相等，那么少的一个放完时，剩下一个数组就可以直接依次放进nums1
最后返回nums1

var merge = function(nums1, m, nums2, n) {
  let m1 = m - 1
  let n1 = n - 1
  let leng = m + n -1
  while(m1 >= 0 && n1 >= 0) {
    nums1[leng--] = nums1[m1] > nums2[n1] ? nums1[m1--] : nums2[n1--]
  }
  while(m1 >= 0) {
    nums1[leng--] = nums1[m1--]
  }
  while(n1 >= 0) {
    nums1[leng--] = nums2[n1--]
  }
  return nums1
};

🔒70.爬楼梯

动态规划解法
核心思维是 meno[i] = meno[i-1] + meno[i-2]
第i个阶梯的次数是前两次的和
与🔒509. 斐波那契数类似

var climbStairs = function(n) {
    let meno = []
    meno[1] = 1
    meno[2] = 2
    for(let i = 3; i <= n; i++) {
        meno[i] = meno[i-1] + meno[i-2]
    }
    return meno[n]
};

🔒13.罗马数字转整数




var romanToInt = function(s) {
    const map = {
        I: 1,
        V: 5,
        X: 10,
        L: 50,
        C: 100,
        D: 500,
        M: 1000,
        IV: 4,
        IX: 9,
        XL: 40,
        XC: 90,
        CD: 400,
        CM: 900
    }
    let result = 0;
    for(let i = 0;i < s.length;) {
        if(i + 1< s.length && map[s.substring(i, i+2)]) {
            result += map[s.substring(i, i+2)];
            i +=2;
        }else {
            result += map[s.substring(i, i+1)];
            i++;
        }
    }
    return result;
};

🔒14. 最长公共前缀




var longestCommonPrefix = function(strs) {
    let res = strs[0];
    if(strs.length === 1) {
        return res;
    }
    for(let i = 1;i < strs.length;i++) {
        while(strs[i].slice(0, res.length) !== res) {
            res = res.slice(0, res.length - 1);
        }
    }
    return res;
};

🔒58.最后一个单词的长度

类似双指针的解法





var lengthOfLastWord = function(s) {
    let end = s.length - 1;
    while(end > 0 && s[end] === ' ') end--;
    if(end < 0) return 0;
    let start = end;
    while(start >= 0 && s[start] !== ' ') start--;
    return end - start;
};

🔒58.最后一个单词的长度

利用indexOf和findIndex两个数组函数实现，因为是按顺序，如果不存在，则在大于这个数的第一个元素前面插入即可。

var searchInsert = function(nums, target) {
    let res = nums.indexOf(target);
    if(res !== -1) {
        return res;
    }
    res = nums.findIndex(item => {
        return item > target;
    })
    return res === -1 ? nums.length : res;
};
